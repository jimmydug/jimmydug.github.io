MySql速学速回

2017-03-31
MySql

Access MySql server

进入数据库系统：
mysql -u user_name -p;
进入数据库系统且指定数据库：
mysql -u user_name -p database_name;

Working with database

创建新数据库:
CREATE DATABASE database_name;
使用某个数据库：
USE database_name;
删除某个数据库：
DROP DATABASE database_name;
查看所有数据库：
SHOW DATABASES;

Working with tables

查看选定数据库下所有表：
SHOW TABLES;
删除某个表：
DROP TABLE table_name;
查看表的各字段详细信息：
DESC table_name [column];
创建新表:
 CREATE TABLE table_name 
(
 pri_key_name INT NOT NULL AUTO_INCREMENT,
 col2_name type(size) NOT NULL,
 col3_name varchar(10) DEFAULT defu_value,
 PRIMARY KEY(pri_key_name)
);
查看创建表时的过程：
SHOW CREATE TABLE table_name;
向表中插入数据(一行或多行)：
列出所有字段：
INSERT INTO table_name 
(col1_name, col2_name, col3_name, 。。。)
VALUES
(col1_value1, col2_value1, col3_value1, 。。。),
(col1_value2, col2_value2, col3_value2, 。。。),
                。。。
                。。。
(col1_value_n, col2_value_n, col3_value_n。。。);


省略所有字段(必须为所有字段赋值)：
INSERT INTO table_name 
VALUES
(col1_value1, col2_value1, col3_value1。。。),
(col1_value2, col2_value2, col3_value2。。。),
                。。。
                。。。
(col1_value_n, col2_value_n, col3_value_n。。。);

为部分指定字段赋值：
INSERT INTO table_name
(col1_name, col3_name, col7_name)
VALUES
(col1_value1, col3_value1, col7_value1),
(col1_value2, col3_value2, col7_value2),
                。。。
                。。。
(col1_value_n, col3_value_n, col7_value_n);

Querying Data

显示表结构：
SHOW COLUMNS FRON table_name;
(=DESC table_name)
显示整个表数据：
SELECT * FROM table_name;
查看符合条件的记录(record):
 SELECT * FROM table_name
 WHERE
 col_x_name = col_x_value
 AND [OR]
 col_y_name = col_y_value
 AND [OR]
 [
 col_name >(< >= <= <>) value;
 col_name IS NULL ;
 (正则表达式，仅文本) 
 col_name LIKE "%inese";
 col_name LIKE "_im";
 col_name BETWEEN 10 AND 30;
 col_name IN (value1,value2,value3, ...);
 (NOT直接放在WHERE,AND,OR后面)
 NOT col_name BETWEEN 1 AND 5;
 NOT col_name LIKE "%iness";
 NOT col_name IN (value1,value2,value3, ...);
]
查看返回的record中指定的field：
 SELECT col_i_name,col_j_name  
 WHERE
 col_x_name = col_x_value
 AND
 col_y_name = col_y_value；
[
 OR 
 col_y_name = col_y_value
]

Correcting the Data

DELETE不能删除单一列的值或某一列所有值，只可用于删除一行或多行。

删除表中所有record：
DELETE FROM table_name;
删除指定的record：
DELETE FROM table_name
WHERE
...
(条件与上SELECT相同)
UPDATE修正已存在的指定record中指定field的值：
UPDATE table_name
SET
col1_name = value1
col2_name = value2
WHERE
...
;
ALTER TABLE为已有表的结构做修改：
ALTER TABLE table_name
ADD COLUMN col_name INT NOT NULL AUTO_INCREMENT
ADD PRIMARY KEy(col_name)(添加主键)
